Normal Button

It's a normal button.

normal info warning danger success

To call the component 

                      
                            <c3m-btn>normal</c3m-btn>
                            <c3m-btn ngClass="info" >info</c3m-btn>
                            <c3m-btn ngClass="warning" >warning</c3m-btn>
                            <c3m-btn ngClass="danger" >danger</c3m-btn>
                            <c3m-btn ngClass="success">success</c3m-btn>

For btn.component.html 

                      
                            <button (click)="OnClick()">
                                <ng-content></ng-content>
                            </button>

For the buttons 

                  
                          c3m-btn button{
                            color: white;
                            padding: 0.5rem 1rem 0.5rem 1rem;
                            border-radius: 3px;
                            background: rgb(105, 105, 105);
                            cursor:pointer;
                            margin-bottom:1.6rem;
                          }
                          c3m-btn button:hover{
                              opacity:0.8;
                          }

For custom colors 

                  
                     c3m-btn.success button{
                          background-color: green;
                      }
                      c3m-btn.info button{
                          background-color: #1773B2;
                      }
                      c3m-btn.warning button{
                          background-color: orange;
                      }
                      c3m-btn.danger button{
                          background-color: orangered;
                      }

To call the component 

                      
                        import {Component, ViewEncapsulation, ElementRef} from '@angular/core';
                            @Component({
                              selector: 'c3m-btn',
                              templateUrl: './btn.component.html',
                              styleUrls: ['./btn.component.css'],
                              encapsulation: ViewEncapsulation.None
                            })
                            export class BtnComponent {

                              constructor(private elementRef: ElementRef) {
                              }

                              public OnClick() {
                                alert('My Class is : ' + this.elementRef.nativeElement.className);
                              }
                            }

Upload Button

It's a file upload button (multiple files)

Choose a file

To call the component 

                      
                            <c3m-btn-file>Choose a file</c3m-btn-file>

For btn-file.component.html 

                      
                            <label for="fileToUpload"><ng-content></ng-content></label>
                            <input (change)= " onChange(file.files) " type= " file " name= " fileToUpload " #file id= " fileToUpload " multiple>
                            <em *ngFor=" let f of files"> { { f.name } } </em>

For the button

                         
                               c3m-btn-file label{
                                    color: white;
                                    padding: 0.5rem 1rem 0.5rem 1rem;
                                    border-radius: 3px;
                                    background: rgb(105, 105, 105);
                                    cursor: pointer;
                                    margin-bottom: 1.6rem;
                                }
                                c3m-btn-file label:hover{
                                    opacity:0.8;
                                }

                                c3m-btn-file input[type="file"]{
                                    position:absolute;
                                    left:-9999px;
                                }

                                c3m-btn-file ul li {
                                    font-family: "Robotoitalic";
                                    color: #676767;
                                    margin-top: 0.3rem;
                                }
                                c3m-btn-file ul li:first-child {
                                    margin-top: 1rem;
                                }

To call the component 

                      
                        import { Component, ViewEncapsulation } from '@angular/core';

                            @Component({
                              selector: 'c3m-btn-file',
                              templateUrl: './btn-file.component.html',
                              styleUrls: ['./btn-file.component.css'],
                              encapsulation: ViewEncapsulation.None
                            })
                            export class BtnFileComponent {
                                files: FileList;

                                onChange(files: FileList) {
                                    this.files = files;
                                }
                            }
